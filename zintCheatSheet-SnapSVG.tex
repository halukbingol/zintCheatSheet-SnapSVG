\subsection{Path Description}
\label{sec:pathdescription}

\begin{tabular}{@{}ll@{}}
	$\sM\ x_{1}\ y_{1}$	&move to point $P_{1}$ at $(x_{1}, y_{1})$\\
	$\sZ$	&close path\\
	$\sL\ x_{2}\ y_{2}$	&line to point $P_{2}$ at $(x_{2}, y_{2})$\\
	$\sH\ x_{2}$	&horizontal line to $P_{2}$ at $(x_{2}, y_{1})$\\
	$\sV\ y_{i}$	&vertical line to $P_{2}$ at $(x_{1}, y_{2})$\\
	$\sC\ x_{1}^{c}\ y_{1}^{c}\ x_{2}^{c}\ y_{2}^{c}\ x_{2}\ y_{2}$
		&curve to $P_{2}$ with control points $P_{1}^{c}$ and $P_{2}^{c}$\\
	$\sS\ x_{2}\ y_{2}\ ??$	&smooth curve to point $P_{2}$ at $(x_{2}, y_{2})$\\
	$\sQ\ x_{1}^{c}\ y_{1}^{c}\ x_{2}\ y_{2}$
		&quadratic Bezier to point $P_{2}$ at $(x_{2}, y_{2})$\\
	$\sT\ ??$	&smooth quadratic Bezier to\\
	$\sA\ r_{x}\ r_{y}\ \beta_{r}\ f_{la}\ f_{s}\ x_{2}\ y_{2}$
		&elliptic arc ending at $(x_{2}, y_{2})$\\
